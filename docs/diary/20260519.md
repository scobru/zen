```js
    gossiped.add(n);
    route.say({ dam: "pex", peers: [PeerRegistry.alt(n)] }, ...); // gossip 1 lần/session
  }
  return;
}
```

## 1. Bug: BOOT peers bị tombstone — infinite reconnect loop

### Triệu chứng

pi và duke có PID cao hơn các VPS relay. Khi kết nối:
1. pi/duke connect outbound đến zen/zen0/zen1
2. AXE dedup (PID-sort): relay có PID thấp hơn giữ outbound của mình → drop outbound của pi/duke
3. Outbound bị close trước khi `root.on("hi")` fire (`met = false`) → `axeGuess++`
4. Sau 5 lần: `_noReconnect = true`, URL bị tombstone
5. BOOT-WATCHDOG clear tombstone sau 30s → lại connect → lại bị tombstone → vòng lặp vô hạn

### Root cause

`axeGuess` và `hiGuess` tombstone checks không phân biệt BOOT peer và peer thường.

```js
// Cũ — tombstone cả BOOT peers
if (peer._isOutbound && !peer.met) {
  peer._axeGuess = (peer._axeGuess || 0) + 1;
  if (peer._axeGuess >= 5) { tombAdd(opt, peer.url); }
}
```

### Fix — `src/websocket.js`

```js
// Mới — BOOT peers không bị tombstone
if (!peer._isBoot && peer._isOutbound && !peer.met) {
  peer._axeGuess = (peer._axeGuess || 0) + 1;
  if (peer._axeGuess >= 5) { peer._noReconnect = true; tombAdd(opt, peer.url); }
}

if (!peer._isBoot && peer._isOutbound && peer.met && peer._openAt &&
    (+new Date() - peer._openAt) < 8000) {
  peer._hiGuess = (peer._hiGuess || 0) + 1;
  if (peer._hiGuess >= 3) { peer._noReconnect = true; tombAdd(opt, peer.url); }
}
```

### Fix — `script/server.js`

BOOT peers được đánh dấu `_isBoot = true` ngay khi khởi tạo và sau mỗi lần WATCHDOG reconnect:

```js
const peerObj = existing || { id: normUrl, url: normUrl, retry: 9 };
peerObj._isBoot = true;  // ← không tombstone
try { route.hi(peerObj); } catch {}
const stored = _initOpt?.peers?.[normUrl];
if (stored) stored._isBoot = true;  // ← giữ sau khi open() copy lại
```

**Commit:** `1cbebd8` — "fix: prevent tombstoning BOOT peers"

## 2. Bug: duke vô hình với các peers khác

### Triệu chứng

`duke thấy các peers nhưng các peers không thấy duke` — duke có UDP fast path đến tất cả node
nhưng không xuất hiện trong `/peers.json` của bất kỳ node nào.

### Điều tra

```
duke domain file: zen.duke.io.vn
duke SSL certs:   không có (no-certs)
duke service env: PORT=8420 (không có DOMAIN=, HTTPS_KEY=)
```

Duke chạy sau **Cloudflare Tunnel** — TLS được terminate bởi Cloudflare, node nội bộ chạy HTTP
trên port 8420. Domain `zen.duke.io.vn` accessible qua `https://zen.duke.io.vn/` (port 443)
nhưng port 8420 bị Cloudflare block.

Vì không có SSL cert local → `opt.key = null` → pex.js tính:
```
scheme = "ws"         (không có key → không wss)
annPort = 8420        (không có PUBLIC_PORT override)
origin = "ws://zen.duke.io.vn:8420/zen"  ← SAI
```

Các VPS nhận URL này, normalize thành `http://zen.duke.io.vn:8420/zen`, thử kết nối →
Cloudflare chặn port 8420 → thất bại.

Verified: `https://zen.duke.io.vn/.well-known/peers.json` → HTTP 200 ✓

### Root cause phân tích

Vấn đề cốt lõi: pex.js không thể biết liệu TLS có đang được terminate externally hay không
chỉ từ `opt.key`. Các deployment phổ biến:

| Deployment | Local cert | Cách announce đúng |
|------------|-----------|---------------------|
| Bare server + certbot | có | `wss://domain:port/zen` |
| Cloudflare Tunnel | không | `wss://domain/zen` (port 443 ẩn) |
| nginx/Caddy reverse proxy | không | `wss://domain/zen` |
| Local HTTP | không | `ws://domain:port/zen` |

### Fix — `lib/pex.js`: tự động probe HTTPS

Thay vì yêu cầu env var `PUBLIC_PORT=443` (thủ công, dễ quên), code tự probe domain khi khởi động:

```js
async function _probeHttps(domain, timeoutMs = 6000) {
  try {
    const ctrl = new AbortController();
    const timer = setTimeout(() => ctrl.abort(), timeoutMs);
    const res = await fetch(`https://${domain}/status`, { signal: ctrl.signal });
    clearTimeout(timer);
    return res.status === 200;
  } catch { return false; }
}
```

Trong `setupRelayPex`, khi `domain && !key`:

```js
if (domain && !key) {
  _probeHttps(domain).then(ok => {
    if (!ok) return;
    const upgraded = `wss://${domain}/zen`;  // port 443 ẩn, chuẩn
    if (upgraded === origin) return;
    if (origin) registry.setSelf(origin);    // absorb URL cũ vào self-set
    origin = upgraded;
    registry.setSelf(origin);
    console.log("[PEX] HTTPS detected — upgraded self-URL to:", origin);
    if (route) { route.say({ dam: "pex", peers: [origin] }); }
  });
}
```

**Cơ chế:** Probe chạy background ~6s sau startup. Nếu domain có HTTPS → tự upgrade scheme.
Không cần env var, không cần cấu hình thủ công. Works cho Cloudflare Tunnel, nginx, Caddy,
bất kỳ TLS terminator nào.

**Log sau fix:**
```
[PEX] HTTPS detected — upgraded self-URL to: wss://zen.duke.io.vn/zen
[WS-ACCEPT] new inbound from: 2407:3640:2328:2041::1   ← zen connects to duke
[WS-ACCEPT] new inbound from: 2405:4802:1bb0:4ba0:...  ← pi connects to duke
[WS-ACCEPT] new inbound from: 2605:a140:2285:6253::1   ← zen1 connects to duke
```

**Commit:** `243bd41` — "fix(pex): auto-detect HTTPS termination by probing domain on startup"

## 3. Bug: `peers.json` hiển thị `"zen.duke.io.vn:"` — thiếu port

### Triệu chứng

```json
{ "peers": ["zen.duke.io.vn:"] }
```

Dấu `:` ở cuối, không có port number.

### Root cause

Code trong `/.well-known/peers.json` endpoint:

```js
const u = new URL(e.url);
return u.hostname + ":" + u.port;  // ← u.port = "" khi port là default
```

`new URL("https://zen.duke.io.vn/zen").port` → `""` (chuỗi rỗng) vì 443 là default port của
HTTPS. Kết quả: `"zen.duke.io.vn:"`.

### Fix — `script/server.js`

```js
const u = new URL(e.url);
const port = u.port || (u.protocol === "https:" ? "443" : "8420");
return u.hostname + ":" + port;
```

**Commit:** `7a886ea` — "fix(server): peers.json port empty for standard-port URLs (wss port 443)"

## 4. Bug: zen0 không thấy pi trong khi tất cả node khác thấy

### Triệu chứng

```
zen.akao.io:  ["zen0", "zen1", "pi:8420", "duke:443"]  ✓
zen0.akao.io: ["zen",  "zen1", "duke:443"]              ✗ thiếu pi
zen1.akao.io: ["zen",  "zen0", "pi:8420", "duke:443"]  ✓
```

### Điều tra

Log zen0 cho thấy pi ĐÃ được discover:
```
[PEX] Discovered peer: https://pi.akao.io:8420/zen
[UDP] Fast path for 0r6RqNzw → 2405:4802:1bb0:4ba0:653e:de5:c829:ed9b:8421 (v6)
```

pi kết nối inbound đến zen0 (UDP fast path hoạt động) nhưng không xuất hiện trong peers.json.

### Root cause

`peers.json` endpoint dùng:
```js
const entries = [
  ...registry.bootEntries(),
  ...registry.confirmedNonBoot(),  // ← chỉ entries có lastOk > 0
];
```

`confirmedNonBoot()` filter: `e.lastOk > 0 && !e.isBoot && !e.isSelf`

`lastOk` được set bởi `registry.confirm()`. Có hai nơi gọi confirm:
1. `root.on("hi")` — khi có `peer.url` (outbound connections) → ✓
2. `mesh.hear["opt"]` wrapper — khi nhận `dam:"opt"` từ inbound peer → chỉ cho BOOT peers!

```js
// Cũ — chỉ confirm BOOT peers khi inbound
mesh.hear["opt"] = function(msg, peer) {
  _origHearOpt.call(this, msg, peer);
  if (...) {
    const ann = PeerRegistry.norm(msg.opt.peers);
    if (ann && registry.isBoot(ann)) {   // ← BUG: bỏ qua pi, duke
      registry.confirm(ann, ...);
    }
  }
};
```

pi và duke là non-BOOT peer. Khi pi kết nối inbound đến zen0:
- AXE dedup có thể drop outbound của zen0 → `root.on("hi")` không fire với `peer.url`
- pi gửi `dam:"opt"` announce URL → `isBoot("https://pi.akao.io:8420/zen")` → false → bỏ qua
- `lastOk` của pi mãi = 0 → không bao giờ xuất hiện trong `confirmedNonBoot()`

### Fix — `script/server.js`

```js
// Mới — confirm TẤT CẢ inbound peers (trừ self)
mesh.hear["opt"] = function(msg, peer) {
  _origHearOpt.call(this, msg, peer);
  if (!msg.ok && msg.opt && typeof msg.opt.peers === "string" && peer && !peer.url && peer.pid) {
    const ann = PeerRegistry.norm(msg.opt.peers);
    if (ann && !registry.isSelf(ann)) {  // ← confirm mọi peer, không phân biệt BOOT
      registry.confirm(ann, { pub: peer.pub || "", pid: peer.pid });
    }
  }
};
```

**Kết quả sau fix:**
```
zen.akao.io:  ["zen0:8420", "zen1:8420", "pi:8420", "duke:443"]  ✓
zen0.akao.io: ["zen:8420",  "zen1:8420", "pi:8420", "duke:443"]  ✓
zen1.akao.io: ["zen:8420",  "zen0:8420", "pi:8420", "duke:443"]  ✓
```

Tất cả 5 node thấy nhau. Pi và duke hiện diện đầy đủ.

**Commit:** `f252fa3` — "fix(server): confirm all inbound peers via dam:opt, not just BOOT peers"

## 5. Fix phụ: deploy.mjs cho pi cần sudo

pi.akao.io dùng user `x` (không phải root) → `systemctl restart zen` cần sudo. Deploy script
cũ dùng `zen restart || systemctl restart zen` — cả hai đều fail trên pi.

Fix: detect theo `username` và inject `sudo -S` khi cần:

```js
function deployCmd(conf) {
  const restart = conf.username === 'root'
    ? 'zen restart 2>&1 || systemctl restart zen 2>&1'
    : `zen restart 2>&1 || echo ${conf.password} | sudo -S systemctl restart zen 2>&1`;
  return [..., restart, ...].join(' && ');
}
```

## Bài học

**1. AXE dedup và BOOT peers không tương thích** — tombstone logic được thiết kế cho peers
thường (bỏ qua peers xấu). BOOT peers phải luôn được reconnect dù AXE drop. Guard `!_isBoot`
là cần thiết ở mọi nơi có tombstone logic.

**2. TLS termination externally là pattern phổ biến** — Cloudflare Tunnel, nginx, Caddy đều
terminate TLS trước khi đến app. Code không nên assume `opt.key` = TLS. Probe thực tế là cách
duy nhất đáng tin cậy để biết domain có HTTPS hay không.

**3. `confirm()` phải được gọi cho mọi inbound connection** — không chỉ BOOT. Bất kỳ peer nào
announce URL của mình qua `dam:"opt"` đều cần được confirm để tồn tại trong `confirmedNonBoot()`.
Pattern guard `isBoot` ban đầu quá restrictive và gây bug invisible peers.

**4. `new URL().port`** trả về chuỗi rỗng cho default ports — không phải `443` hay `80`. Luôn
fallback khi build host:port string từ URL object.
